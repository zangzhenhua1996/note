    numpy的深拷贝,浅拷贝¶

    import numpy as np

    arr1 = np.array([1,2,3])

    arr2 = arr1#arr1,arr2共享一块内存，浅拷贝

print(arr1)
print(arr2)

    arr3 = arr1.copy()#深拷贝这里就相当于普通变量的deepcopy

    arr3[0] = 10
print(arr1)
print(arr3)

    [5 2 3]
